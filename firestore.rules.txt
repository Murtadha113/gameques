rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // بيانات اللعبة (الفئات والأسئلة) - الكل يقدر يقرأ، بس الأدمن بس يقدر يعدل
    match /couplesGame/data {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == 'WyzmPfdR9mNd6XKyc1jQzpMCb2x2';
    }

    // الأسئلة الخاصة اللي يضيفها اللاعبين لأنفسهم (والصورة المرفقة كنص base64)
    // - أي زائر (حتى بدون تسجيل، عبر Anonymous Auth) يقدر يضيف سؤال باسمه (ownerUid) الخاص فيه، ويصير جزء من أسئلته هو بس فورًا بدون موافقة أدمن
    // - الأدمن يقدر يشوف/يحذف الكل للمراجعة، وصاحب السؤال نفسه يقدر يشوف/يحذف أسئلته هو بس
    match /pendingQuestions/{docId} {
      allow create: if request.auth != null
                    && request.resource.data.ownerUid == request.auth.uid
                    && request.resource.data.keys().hasAll(['name', 'text', 'catId', 'mode', 'ownerUid', 'createdAt'])
                    && request.resource.data.name is string && request.resource.data.name.size() > 0 && request.resource.data.name.size() < 60
                    && request.resource.data.text is string && request.resource.data.text.size() > 0 && request.resource.data.text.size() < 500
                    && (!('photoBase64' in request.resource.data) || request.resource.data.photoBase64 == null
                        || (request.resource.data.photoBase64 is string && request.resource.data.photoBase64.size() < 950000));
      allow update: if false;
      allow read, delete: if request.auth != null
                           && (request.auth.uid == 'WyzmPfdR9mNd6XKyc1jQzpMCb2x2' || request.auth.uid == resource.data.ownerUid);
    }

    // علامات "تم لعب هذا السؤال" - ما فيها نص الإجابة، بس تستخدم لمنع تكرار نفس السؤال لنفس الحساب
    match /playedQuestions/{docId} {
      allow create: if request.auth != null
                    && request.resource.data.ownerUid == request.auth.uid
                    && request.resource.data.keys().hasAll(['ownerUid', 'catId', 'questionText', 'createdAt']);
      allow read: if request.auth != null
                  && (request.auth.uid == 'WyzmPfdR9mNd6XKyc1jQzpMCb2x2' || request.auth.uid == resource.data.ownerUid);
      allow update: if false;
      allow delete: if request.auth != null && request.auth.uid == 'WyzmPfdR9mNd6XKyc1jQzpMCb2x2';
    }

    // الإجابات الفعلية على الأسئلة (إجابة الطرفين مع بعض بنفس المستند) - خاصة بين صاحب الحساب والأدمن بس
    match /answeredQuestions/{docId} {
      allow create: if request.auth != null
                    && request.resource.data.ownerUid == request.auth.uid
                    && request.resource.data.keys().hasAll(['ownerUid', 'catId', 'mode', 'questionText', 'player1Name', 'player1Answer', 'player2Name', 'player2Answer', 'createdAt'])
                    && request.resource.data.player1Name is string && request.resource.data.player1Name.size() > 0 && request.resource.data.player1Name.size() < 60
                    && request.resource.data.player2Name is string && request.resource.data.player2Name.size() > 0 && request.resource.data.player2Name.size() < 60
                    && request.resource.data.player1Answer is string && request.resource.data.player1Answer.size() > 0 && request.resource.data.player1Answer.size() < 2000
                    && request.resource.data.player2Answer is string && request.resource.data.player2Answer.size() > 0 && request.resource.data.player2Answer.size() < 2000;
      allow update: if false;
      allow read, delete: if request.auth != null
                           && (request.auth.uid == 'WyzmPfdR9mNd6XKyc1jQzpMCb2x2' || request.auth.uid == resource.data.ownerUid);
    }
  }
}
